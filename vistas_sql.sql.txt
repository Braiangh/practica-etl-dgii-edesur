CREATE OR REPLACE VIEW vista_rnc_regimen_simplificado AS
SELECT 
    columna_1 AS rnc_cedula, 
    columna_2 AS razon_social, 
    columna_6 AS regimen_pago
FROM rnc_contribuyentes
WHERE columna_6 LIKE '%SIMPLIFICADO%';

CREATE OR REPLACE VIEW vista_rnc_regimen_normal AS
SELECT 
    columna_1 AS rnc_cedula, 
    columna_2 AS razon_social, 
    columna_6 AS regimen_pago
FROM rnc_contribuyentes
WHERE columna_6 = 'NORMAL';
